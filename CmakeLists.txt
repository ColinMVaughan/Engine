cmake_minimum_required(VERSION 3.0)
project(Application)

add_subdirectory(ECS)
add_subdirectory(GLRenderer)

target_include_directories(ECS PUBLIC 
		$<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/Include>
		$<INSTALL_INTERFACE:Include>
		)
		
add_library(Application Application/Include/Application.h Application/Src/Main.cpp Application/Src/Application.cpp)
target_link_libraries(Application ECS GLRenderer)

install(TARGETS Application 
		ARCHIVE DESTINATION Lib
		LIBRARY DESTINATION Lib
		INCLUDES DESTINATION Include)
		
install(FILES Application.h DESTINATION Include/Application)
